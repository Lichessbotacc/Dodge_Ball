<script>
const canvas = document.getElementById("game");
const ctx = canvas.getContext("2d");

function resize(){
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
}
resize();
window.addEventListener("resize", resize);

// Players
const p1 = { x: 200, y: 300, r: 18, color:"#00ff88" };
const p2 = { x: 500, y: 300, r: 18, color:"#4488ff" };

// Balls
let balls = [];

function createBall(){
    balls.push({
        x: Math.random()*canvas.width,
        y: Math.random()*canvas.height,
        vx:(Math.random()*4+2)*(Math.random()<0.5?-1:1),
        vy:(Math.random()*4+2)*(Math.random()<0.5?-1:1),
        r: 18
    });
}

for(let i=0;i<4;i++) createBall();

// JOYSTICK DATA
let joy1 = { dx:0, dy:0, active:false, id:null };
let joy2 = { dx:0, dy:0, active:false, id:null };

// joystick helper
function setupJoystick(areaId, stickId, joy){
    const area = document.getElementById(areaId);
    const stick = document.getElementById(stickId);

    area.addEventListener("touchstart",(e)=>{
        joy.active = true;
        joy.id = e.targetTouches[0].identifier;
    });

    area.addEventListener("touchmove",(e)=>{
        for(let t of e.touches){
            if(t.identifier === joy.id){

                const rect = area.getBoundingClientRect();
                const x = t.clientX - rect.left - rect.width/2;
                const y = t.clientY - rect.top - rect.height/2;

                joy.dx = x / 50;
                joy.dy = y / 50;

                stick.style.transform =
                    `translate(${x}px, ${y}px)`;
            }
        }
    });

    area.addEventListener("touchend",()=>{
        joy.active = false;
        joy.dx = 0;
        joy.dy = 0;
        stick.style.transform = "translate(0px,0px)";
    });
}

setupJoystick("joy1","stick1",joy1);
setupJoystick("joy2","stick2",joy2);

// Update
function update(){

    // player movement
    p1.x += joy1.dx * 5;
    p1.y += joy1.dy * 5;

    p2.x += joy2.dx * 5;
    p2.y += joy2.dy * 5;

    // bounds
    [p1,p2].forEach(p=>{
        p.x = Math.max(p.r, Math.min(canvas.width-p.r,p.x));
        p.y = Math.max(p.r, Math.min(canvas.height-p.r,p.y));
    });

    // balls
    for(let b of balls){

        b.x += b.vx;
        b.y += b.vy;

        if(b.x < 0 || b.x > canvas.width) b.vx *= -1;
        if(b.y < 0 || b.y > canvas.height) b.vy *= -1;

        checkCollision(p1,b);
        checkCollision(p2,b);
    }
}

function checkCollision(p,b){
    const dx = p.x - b.x;
    const dy = p.y - b.y;
    const d = Math.sqrt(dx*dx + dy*dy);

    if(d < p.r + b.r){
        alert("GAME OVER!");
        location.reload();
    }
}

// Draw
function draw(){
    ctx.clearRect(0,0,canvas.width,canvas.height);

    // balls
    for(let b of balls){
        ctx.beginPath();
        ctx.arc(b.x,b.y,b.r,0,Math.PI*2);
        ctx.fillStyle="red";
        ctx.fill();
    }

    // players
    [p1,p2].forEach(p=>{
        ctx.beginPath();
        ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
        ctx.fillStyle=p.color;
        ctx.fill();
    });
}

function loop(){
    update();
    draw();
    requestAnimationFrame(loop);
}
loop();
